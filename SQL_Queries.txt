create table Author(
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
    first_name varchar(100) not null ,
    last_name varchar(100) not null
);

create table Book (
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
    name varchar(100) not null ,
    pageCount int not null ,
    author_id int references author(id)
);
